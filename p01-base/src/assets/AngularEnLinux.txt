#Creo una carpeta en mi equipo para mis proyectos

1. mkdir GIT/



#Instalo git

1. sudo apt-get update

2. sudo apt-get install git



#Creo un repositorio en GitHub y lo clono en mi directorio

1. cd GIT/

2. git clone url_mi_repositorio



#Instalar nodejs

1. sudo apt-get install curl -y

2. curl -sL https://deb.nodesource.com/setup_6.x | sudo -E bash -

3. sudo apt-get install nodejs



#Check versiones

1. nodejs -v

2. npm -v

#Instalar angular cli
1. sudo npm install -g angular-cli



#Check versión angular cli

1. ng -v



#Inicializar angular con primer proyecto

1. ng new nombre_de_mi_proyecto



#Levantar servidor

1. ng serve



#Hacer cambios

1. Abro Visual Studio Code

2. Edito src/app/app.components.ts (Le pongo una frase)

3. Navegador: localhost: 4200 (para ver los cambios)



#Git en Visual Studio Code

1. Icono de control de código fuente

2. Confirmar (icono tick) (guarda en local)

3. Publicar Rama ó Sincronizar (sube a repositorio GitHub)
